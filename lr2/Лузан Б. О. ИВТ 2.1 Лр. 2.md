### Математическая модель:

### Список идентификаторов:

<table>
  <tr>
    <th>имя</th>

	<th>смысл</th>
  </tr>

    <td>n</td>
    <td>int</td>

### Код программы:






```C
#include <stdio.h>

#include <stdlib.h>

#include <malloc.h>

  

int main(void)

{

    int n;

    printf("imput number of elements ");

    scanf("%d", &n);

    double *p = malloc(n * sizeof(double));

    int i;

  

    for(i = 0;i <= n-1; i++)

    {

        scanf("%lf", p+i);

    }

  

    for(i = 0;i <= n-1; i++)

    {

        printf("%lf\n", *(p+i));

    }

    free(p);

  
  

    return 0;

}
```

    }
